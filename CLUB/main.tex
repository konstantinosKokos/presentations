\section{(Meta)science}

\begin{frame}{Neopositivism; a primer}

		\alt<4->{
			\centering
			\framebox{\textsc{Number} witnesses \textsc{Truth}.}
		}{
		\[
		\boxed{
			\textsc{Truth} \equiv \mathbb{1}
		}
		\]
		\smaller
		\vfill
		\uncover<2-3>{
			Scientific methodology:
			\begin{enumerate}
				\item observe
				\item theorize
				\item fit
				\item evaluate
				\item accept/reject/refine | repeat
			\end{enumerate}\vfill
			
			\uncover<3->{
					Theories are refutable; conclusions are not.\\
					Reproducibility ensures results are objective, neutral \& \textit{universal}.\vfill
			}
		}
		}
					
\end{frame}

\begin{frame}{The problem with language}
	\smaller
	Where it all falls apart
	\begin{enumerate}
		\item TL: lack of computational foundations
		\uncover<2->{
		\item CL: theories either too narrow or just suck
		\uncover<3->{
		\item NLP: engineering trumps theory
		\uncover<4->{
		\item[]  $\implies$ \alert{theory} iteration $\leadsto$ \alert{model} iteration
		}}}
	\end{enumerate}\vfill
	\visible<4->{
	\centering
	Today, CL papers come in two main flavors: \textbf{benchmarks} \& \textbf{models} -- \\
	-- complementary facets of the same one thing: \textbf{tables with numbers}.
	}
\end{frame}

\begin{frame}{Models\\
\quad \smaller (large or otherwise, language or otherwise) \visible<3->{are \textbf{not} a substitute to theory}}
	\alt<3->{
		\centering
		\framebox{\textsc{Number} witnesses \st{\textsc{Truth}} \alert{\textsc{Performance}}.}
	}{
		\smaller
		\begin{itemize}
			\item theory-neutral
			\item infallible (other than in performance)
			\item[]  $\implies$ have \alert{nothing} to say (other than know-how)
			\item[]
			\uncover<2->{
			\item language reduced to ``tasks`
			\item ``understanding`` $:= \sum\text{task}$
			\item[]  $\implies$ ``understanding`` \alert{without understanding}
			}
		\end{itemize}
	}
\end{frame}

\begin{frame}{Epistemic Inflation\\
\quad \smaller \visible<-3>{\alt<2->{2}{1}/2: \alt<2->{Effects}{Causes}}
}
	\alt<4->{
		\centering
		\framebox{\textsc{Number} witnesses \st{\textsc{Truth}} \st{\textsc{Performance}}  \alert{\textsc{Nothing}}.}
	}{
		\smaller
		\alt<2->{
			\begin{itemize}
				\item ``progress'' $\leadsto$ leaderboarding competition
				\item performative sales-pitch aesthetics
				\item low-hanging-fruit agendas
				\item knowledge recycling, citation amnesia
				\item less replication
				\visible<3->{
				\item[] $\implies$ more numbers $\leadsto$ \alert{less validity}
				}
			\end{itemize}
		}{
			\begin{itemize}
				\item model iteration is easier than theory iteration
				\item performance is a prerequisite for publication
				\item publication is a prerequisite for employment / status / \dots 
				\item[]  $\implies$ extra incentives for \alert{more numbers}
			\end{itemize}
		}
	}
\end{frame}

\begin{frame}{... and why it's bad}
	\smaller
	The promise: {\smaller \textcolor{gray!90}{(read: neoliberal narrative)}}
	\begin{itemize}
		\item academia produces knowledge
		\item industry actualizes knowledge
		\item academia scrutinizes industry	
	\end{itemize}
	
	\uncover<2->{
	\hfill \textcolor{gray!90}{\textit{implicit condition: pursuit of knowledge $\parallel$ big tech interests}}
	}
	\vfill

	\uncover<3->{	
	In reality:
	\begin{itemize}
		\item industry \st{actualizes} \alert{predates} knowledge
		\uncover<4->{
		\item corporate resources $\leadsto$  means of scientific \st{empowerment} \alert{coercion}
		\uncover<5->{
		\item surviving ideas $\leadsto$ \st{socially relevant tools} \alert{economically viable products}}
		\visible<6->{
		\item[]
		\item[] \hfil \framebox{$\implies$ scientist $\leadsto$ \alert{big tech spokesperson}}
		}}
	\end{itemize}
	}
\end{frame}
